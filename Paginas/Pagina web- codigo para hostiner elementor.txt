        function animateCounters() {
            const counters = document.querySelectorAll('.stat-card');
            counters.forEach(card => {
                const numberEl = card.querySelector('.stat-number');
                const format = card.dataset.format;
                const targetText = numberEl.textContent;
                let target = parseFloat(targetText.replace(/[^\d.]/g, ''));
                const hasDecimal = targetText.includes('.');
                let current = 0;
                const duration = 2000;
                const increment = target / (duration / 16);
                
                const timer = setInterval(() => {
                    current += increment;
                    if (current >= target) {
                        current = target;
                        clearInterval(timer);
                    }
                    let displayNum = hasDecimal ? current.toFixed(1) : Math.floor(current);
                    displayNum = displayNum.toLocaleString();
                    
                    let displayText = displayNum;
                    if (format === 'percent') {
                        displayText = displayNum + '%';
                    } else if (format === 'billion') {
                        displayText = '$' + displayNum + 'B';
                    }
                    numberEl.textContent = displayText;
                }, 16);
            });
        }
